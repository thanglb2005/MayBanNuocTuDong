Class Diagram — Hệ Thống Máy Bán Nước Tự Động

Patterns: Singleton (VendingMachine) + State (VendingMachineState)
Nguyên tắc: Business logic only — không UI/MVC/Framework. Behavior nằm trong State, VendingMachine chỉ delegate.

1. Class Diagram — Business Domain

classDiagram
    direction TB

    %% ==================== PRODUCT HIERARCHY ====================
    class Beverage {
        <<abstract>>
        -String id
        -String name
        -double price
        +getId() String
        +getName() String
        +getPrice() double
        +getDescription() String*
    }
    class Tea {
        -String flavor
        +getDescription() String
    }
    class SoftDrink {
        -String brand
        +getDescription() String
    }
    class MineralWater {
        -double volume
        +getDescription() String
    }
    Beverage <|-- Tea
    Beverage <|-- SoftDrink
    Beverage <|-- MineralWater

    %% ==================== INVENTORY ====================
    class InventoryItem {
        -Beverage beverage
        -int quantity
        +isAvailable() boolean
        +reduce() boolean
        +addStock(amount) void
    }
    class Inventory {
        -List~InventoryItem~ items
        +addBeverage(beverage, qty) void
        +findItem(id) InventoryItem
        +reduceStock(id) boolean
        +isAvailable(id) boolean
        +hasAnyStock() boolean
        +getAvailableItems() List~InventoryItem~
    }

    %% ==================== PAYMENT ====================
    class PaymentMethod {
        <<enumeration>>
        COD
        PAYOS
        +getDisplayName() String
    }

    %% ==================== TRANSACTION ====================
    class Transaction {
        -Beverage beverage
        -PaymentMethod method
        -LocalDateTime timestamp
        +Transaction(beverage, method)
        +calculateAmount() double
    }

    %% ==================== STATE (interface) ====================
    class VendingMachineState {
        <<interface>>
        +setContext(machine) void
        +selectProduct(beverageId) void
        +processPayment(method) void
        +dispense() void
        +getStateName() String
    }

    %% ==================== CONCRETE STATES ====================
    class ReadyState {
        -VendingMachine context
        +selectProduct(beverageId) void
    }
    class ProcessingPaymentState {
        -VendingMachine context
        -Beverage selectedBeverage
        +processPayment(method) void
    }
    class DispensingState {
        -VendingMachine context
        -Beverage selectedBeverage
        -PaymentMethod method
        +dispense() void
    }
    class OutOfStockState {
        -VendingMachine context
    }

    VendingMachineState <|.. ReadyState
    VendingMachineState <|.. ProcessingPaymentState
    VendingMachineState <|.. DispensingState
    VendingMachineState <|.. OutOfStockState

    %% ==================== SINGLETON + CONTEXT ====================
    class VendingMachine {
        -VendingMachine instance$
        -VendingMachineState state
        -Inventory inventory
        -List~Transaction~ transactions
        -VendingMachine()
        +getInstance()$ VendingMachine
        +changeState(state) void
        +selectProduct(beverageId) void
        +processPayment(method) void
        +dispense() void
        +addTransaction(tx) void
        +getInventory() Inventory
        +getTransactions() List~Transaction~
    }

    %% ==================== RELATIONSHIPS ====================
    VendingMachine o-- "1" VendingMachineState : state
    VendingMachine *-- "1" Inventory : inventory
    VendingMachine *-- "*" Transaction : transactions

    Inventory *-- "*" InventoryItem : items
    InventoryItem --> "1" Beverage : beverage

    Transaction --> "1" Beverage : beverage
    Transaction --> "1" PaymentMethod : method

    ReadyState --> VendingMachine : context
    ProcessingPaymentState --> VendingMachine : context
    DispensingState --> VendingMachine : context
    OutOfStockState --> VendingMachine : context

    ProcessingPaymentState --> "1" Beverage : selectedBeverage
    DispensingState --> "1" Beverage : selectedBeverage
    DispensingState --> "1" PaymentMethod : method

3. State Pattern

classDiagram
    direction LR

    class VendingMachine {
        <<Context>>
        -VendingMachineState state
        +changeState(state) void
        +selectProduct(id) void
        +processPayment(method) void
        +dispense() void
    }

    class VendingMachineState {
        <<interface>>
        +setContext(machine) void
        +selectProduct(id) void
        +processPayment(method) void
        +dispense() void
        +getStateName() String
    }

    class ConcreteStates {
        <<Ready | ProcessingPayment | Dispensing | OutOfStock>>
        -VendingMachine context
        +setContext(machine) void
        +selectProduct(id) void
        +processPayment(method) void
        +dispense() void
        +getStateName() String
    }

    VendingMachine o-- "1" VendingMachineState : state
    VendingMachineState <|.. ConcreteStates
    ConcreteStates --> VendingMachine : context

    note for VendingMachine "this.state = state\nstate.setContext(this)"
    note for ConcreteStates "// transition example\nstate = new ReadyState()\ncontext.changeState(state)"

4. State Transition Diagram

stateDiagram-v2
    [*] --> Ready
    Ready --> ProcessingPayment : selectProduct()
    ProcessingPayment --> Dispensing : processPayment() success
    ProcessingPayment --> Ready : cancel / payment fail
    Dispensing --> Ready : dispense() stock remaining
    Dispensing --> OutOfStock : dispense() all stock depleted
    OutOfStock --> Ready : addBeverage()
